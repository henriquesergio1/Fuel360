
# ===== Estágio 1: Build =====
FROM node:18-alpine AS builder

WORKDIR /app

# Copia o package.json e o package-lock.json (se existir)
COPY package*.json ./

# Instala as dependências
RUN npm install

# Copia o resto dos arquivos do projeto
COPY . .

# Limpa builds anteriores para evitar contaminação
RUN rm -rf dist

# Executa o build de produção (os scripts no package.json criarão o diretório dist)
RUN npm run build:prod

# ===== Estágio 2: Produção =====
FROM nginx:alpine

# Copia a configuração personalizada do Nginx
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Copia os arquivos estáticos compilados do estágio de build
COPY --from=builder /app/dist/ /usr/share/nginx/html/dist/

# Copia o arquivo HTML de produção
COPY --from=builder /app/index.prod.html /usr/share/nginx/html/index.html

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
